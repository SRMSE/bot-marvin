/**
    Provides absolute path of the file, from where it is called.
    @static
    @param {String} dirname - pass the __dirname
    @returns {String}
*/
Proto.getAbsolutePath = function(dirname) {
    var temp = dirname.split("/");
    temp.pop();
    var parent_dir = temp.join("/");
    return parent_dir;
}
